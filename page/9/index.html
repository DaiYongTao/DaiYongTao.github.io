#二、示例剖析：

#三、解题思路：

#四、小结

func isPowerOfThree(_ n: Int) -> Bool {

if n <= 0 {
    return false
}

// 将整数转换为浮点型
let max = Double(Int.max)

// 求出Int范围内最大的3的幂 -- 39
let k = Int(log(max) / log(3.0))

// 得到Int范围内最大的3的幂次方，幂=k

let b: Int = Int(pow(3.0, Double(k)))

// 如果n是3的幂次方，那么b必定能整除n
return b % n == 0

}
