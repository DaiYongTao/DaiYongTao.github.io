350. 两个数组的交集 II

Posted on 2018-09-20 | Edited on 2018-12-26 | In leetCode题库笔记

#一、题目原型：

给定两个数组，编写一个函数来计算它们的交集。

#二、题目意思剖析：

#三、解题思路：
1.遍历两个数组，如果发现有元素相同，就用新数组把它存起来。然后，把原来的两个数组里的这个元素设置为Int.max，下一次遍历就不会重复了。
耗时

func intersect(_ nums1: [Int], _ nums2: [Int]) -> [Int] {
if nums1.count == 0 || nums2.count == 0 {
return []
}

var nums: [Int] = []
var mutNums1 = nums1
var mutNums2 = nums2

for i in 0..<mutNums1.count {

    let subNum1: Int = mutNums1[i]
    for j in 0..<mutNums2.count {
        let subNum2: Int = mutNums2[j]
        if (subNum1 == subNum2) {
            nums.append(subNum1)
            mutNums1[i] = Int.max
            mutNums2[j] = Int.max
            break
        }
    }
}
return nums

}
